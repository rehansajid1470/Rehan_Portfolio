Web Development Bug Fixes

Bug #1: Insecure Password Storage

Buggy Implementation 
@app.route('/api/users', methods=['POST'])
def create_user():
    data = request.json
    username = data['username']
    email = data['email']
    password = data['password']  # BUG: Plaintext password
    
    users[username] = {
        'email': email,
        'password': password  # Storing plaintext password!
    }
    
    return jsonify({'status': 'success', 'message': f'User {username} created'})

The bug: Storing passwords in plaintext is a major security vulnerability. If the database is ever compromised, all user passwords are exposed.

Fixed Implementation 
from werkzeug.security import generate_password_hash

@app.route('/api/users', methods=['POST'])
def create_user():
    data = request.json
    username = data['username']
    email = data['email']
    password = data['password']
    
    # FIXED: Hash the password before storing
    password_hash = generate_password_hash(password)
    
    users[username] = {
        'email': email,
        'password_hash': password_hash  # Store hash instead of plaintext
    }
    
    return jsonify({'status': 'success', 'message': f'User {username} created'})

The fix: Using Werkzeug's password hashing utilities to securely store password hashes rather than plaintext passwords.

Impact & Explanation

Storing plaintext passwords has serious security implications:

  • Makes user accounts immediately vulnerable in case of a data breach
  • Violates security best practices and potentially legal requirements (GDPR, etc.)
  • Compromises user trust and can lead to identity theft
  • May expose users across multiple services if they reuse passwords

Bug #2: Missing Input Validation

Buggy Implementation 
@app.route('/api/users/', methods=['GET'])
def get_user(username):
    # BUG: No validation if user exists
    user_data = users[username]  # KeyError if username doesn't exist
    
    # Remove sensitive data
    user_info = {
        'username': username,
        'email': user_data['email']
    }
    
    return jsonify(user_info)

The bug: Accessing the users dictionary without checking if the key exists first. This will raise a KeyError if the username doesn't exist.

Fixed Implementation 
@app.route('/api/users/', methods=['GET'])
def get_user(username):
    # FIXED: Proper validation and error handling
    if username not in users:
        return jsonify({
            'status': 'error',
            'message': 'User not found'
        }), 404  # Return proper HTTP 404 status code
    
    user_data = users[username]
    
    # Remove sensitive data
    user_info = {
        'username': username,
        'email': user_data['email']
    }
    
    return jsonify(user_info)

The fix: Validating that the username exists before attempting to access it, and returning a proper HTTP 404 status code if not found.

Bug #3: Insecure Direct Object Reference (IDOR)

Buggy Implementation 
# User sessions with simplified auth
user_sessions = {}

@app.route('/api/users//profile', methods=['GET'])
def get_user_profile(username):
    # BUG: No authentication check
    # Anyone can access any user's profile information
    
    if username not in users:
        return jsonify({'error': 'User not found'}), 404
    
    user_data = users[username]
    
    profile = {
        'username': username,
        'email': user_data['email'],
        'settings': user_data.get('settings', {})
    }
    
    return jsonify(profile)

The bug: No authentication or authorization checks before returning user data. Anyone can access any user's profile by knowing or guessing the username.

Fixed Implementation 
from flask import Flask, request, jsonify, session
from functools import wraps

# Authentication decorator
def login_required(f):
    @wraps(f)
    def decorated_function(*args, **kwargs):
        # Check if user is logged in
        if 'current_user' not in session:
            return jsonify({'error': 'Authentication required'}), 401
        return f(*args, **kwargs)
    return decorated_function

@app.route('/api/users//profile', methods=['GET'])
@login_required
def get_user_profile(username):
    # FIXED: Authentication + authorization check
    current_user = session['current_user']
    
    # Only allow users to access their own profiles
    if current_user != username:
        return jsonify({'error': 'Unauthorized access'}), 403
    
    if username not in users:
        return jsonify({'error': 'User not found'}), 404
    
    user_data = users[username]
    
    profile = {
        'username': username,
        'email': user_data['email'],
        'settings': user_data.get('settings', {})
    }
    
    return jsonify(profile)

The fix: Adding authentication with a login_required decorator and authorization check to ensure users can only access their own profiles.

Complete Fixed Implementation (Flask User API)

from flask import Flask, request, jsonify, session
from werkzeug.security import generate_password_hash, check_password_hash
from functools import wraps
import os
import secrets

app = Flask(__name__)
app.secret_key = secrets.token_hex(16)  # Secure secret key for sessions

# In-memory user storage (would be a database in production)
users = {}

# Authentication decorator
def login_required(f):
    @wraps(f)
    def decorated_function(*args, **kwargs):
        if 'current_user' not in session:
            return jsonify({'error': 'Authentication required'}), 401
        return f(*args, **kwargs)
    return decorated_function

@app.route('/api/users', methods=['POST'])
def create_user():
    try:
        data = request.json
        
        # Validate required fields
        required_fields = ['username', 'email', 'password']
        for field in required_fields:
            if field not in data:
                return jsonify({
                    'status': 'error',
                    'message': f'Missing required field: {field}'
                }), 400
        
        username = data['username']
        email = data['email']
        password = data['password']
        
        # Check if username already exists
        if username in users:
            return jsonify({
                'status': 'error',
                'message': 'Username already exists'
            }), 409
        
        # Validate email format (basic check)
        if '@' not in email or '.' not in email:
            return jsonify({
                'status': 'error',
                'message': 'Invalid email format'
            }), 400
        
        # Password complexity check (basic example)
        if len(password) < 8:
            return jsonify({
                'status': 'error',
                'message': 'Password must be at least 8 characters'
            }), 400
        
        # Hash the password
        password_hash = generate_password_hash(password)
        
        # Store user data
        users[username] = {
            'email': email,
            'password_hash': password_hash,
            'settings': {}
        }
        
        return jsonify({
            'status': 'success',
            'message': f'User {username} created successfully'
        })
    
    except Exception as e:
        app.logger.error(f"Error creating user: {str(e)}")
        return jsonify({
            'status': 'error',
            'message': 'An unexpected error occurred'
        }), 500

@app.route('/api/login', methods=['POST'])
def login():
    try:
        data = request.json
        
        if 'username' not in data or 'password' not in data:
            return jsonify({
                'status': 'error',
                'message': 'Username and password required'
            }), 400
        
        username = data['username']
        password = data['password']
        
        if username not in users:
            return jsonify({
                'status': 'error',
                'message': 'Invalid username or password'
            }), 401
        
        user_data = users[username]
        
        # Verify password
        if not check_password_hash(user_data['password_hash'], password):
            return jsonify({
                'status': 'error',
                'message': 'Invalid username or password'
            }), 401
        
        # Set session
        session['current_user'] = username
        
        return jsonify({
            'status': 'success',
            'message': 'Login successful'
        })
    
    except Exception as e:
        app.logger.error(f"Login error: {str(e)}")
        return jsonify({
            'status': 'error',
            'message': 'An unexpected error occurred'
        }), 500

@app.route('/api/users/', methods=['GET'])
@login_required
def get_user(username):
    current_user = session['current_user']
    
    # Users can only access their own information
    if current_user != username:
        return jsonify({
            'status': 'error',
            'message': 'Unauthorized access'
        }), 403
    
    if username not in users:
        return jsonify({
            'status': 'error',
            'message': 'User not found'
        }), 404
    
    user_data = users[username]
    
    # Remove sensitive data
    user_info = {
        'username': username,
        'email': user_data['email'],
        'settings': user_data.get('settings', {})
    }
    
    return jsonify(user_info)

@app.route('/api/logout', methods=['POST'])
def logout():
    session.pop('current_user', None)
    return jsonify({
        'status': 'success',
        'message': 'Logged out successfully'
    })

if __name__ == '__main__':
    # In production, use HTTPS and proper session configuration
    app.run(debug=False)

Best Practices for Web Development

Always Hash Passwords

Never store plaintext passwords. Use a modern password hashing library.

# Good practice
from werkzeug.security import generate_password_hash

password_hash = generate_password_hash(password)
# Store password_hash in database

Implement Proper Authentication

Use decorators to protect routes that require authentication.

# Create a decorator
def login_required(f):
    @wraps(f)
    def decorated_function(*args, **kwargs):
        if 'user_id' not in session:
            return redirect(url_for('login'))
        return f(*args, **kwargs)
    return decorated_function

# Use it on protected routes
@app.route('/profile')
@login_required
def profile():
    # Only accessible to logged-in users
    pass

Validate All User Input

Never trust user input. Always validate before processing.

# Validate form data
if not all([name, email, message]):
    return "All fields are required", 400
    
# Validate email format (basic example)
if '@' not in email or '.' not in email:
    return "Invalid email format", 400
    
# Validate data types
try:
    user_id = int(user_id_str)
except ValueError:
    return "Invalid user ID", 400

Use Proper HTTP Status Codes

Return appropriate HTTP status codes for different situations.

# Success
return jsonify(data), 200  # OK

# Created new resource
return jsonify(new_item), 201  # Created

# Bad request (client error)
return jsonify({"error": "Invalid data"}), 400

# Unauthorized
return jsonify({"error": "Login required"}), 401

# Forbidden
return jsonify({"error": "Not allowed"}), 403

# Not found
return jsonify({"error": "Resource not found"}), 404

# Server error
return jsonify({"error": "Server error"}), 500

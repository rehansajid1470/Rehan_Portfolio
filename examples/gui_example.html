Bug Fixes

Bug #1: UI Freezing Due to Improper Update Method

Buggy Implementation 
def update_counter(self):
    while True:
        current_time = time.strftime("%H:%M:%S")
        self.time_label.config(text=current_time)
        time.sleep(1)  # This will freeze the GUI

The bug: Using an infinite loop with time.sleep() inside a GUI method will freeze the application's main thread, making it unresponsive.

Fixed Implementation 
def update_time(self):
    current_time = time.strftime("%H:%M:%S")
    self.time_label.config(text=current_time)
    # Schedule the next update after 1000ms (1 second)
    self.root.after(1000, self.update_time)

The fix: Using Tkinter's after() method to schedule updates, allowing the main event loop to remain responsive.

Impact & Explanation

This bug makes the entire application freeze as soon as it starts. The issue is related to how GUI event loops work:

  • Tkinter's main thread must remain free to process user interactions
  • Blocking operations (like infinite loops with sleep) prevent the event loop from running
  • The correct approach uses the built-in scheduling mechanism that works with the event loop

Bug #2: Improper Button Command Binding

Buggy Implementation 
add_button = tk.Button(entry_frame, text="Add Task", 
                      bg="#2ecc71", fg="white", 
                      font=("Arial", 10, "bold"), 
                      command=self.add_task())

The bug: The command parameter calls the method immediately during widget creation instead of when the button is clicked.

Fixed Implementation 
add_button = tk.Button(entry_frame, text="Add Task", 
                      bg="#2ecc71", fg="white", 
                      font=("Arial", 10, "bold"), 
                      command=self.add_task)

The fix: Passing the method reference without calling it, allowing Tkinter to call it when the button is clicked.

Impact & Explanation

This bug causes two critical issues:

  • The add_task() method is called immediately during interface creation, not when the button is clicked
  • The button's command is set to the return value of add_task() (likely None), making the button non-functional
  • This is a common mistake when assigning callbacks in GUI programming

import tkinter as tk
from tkinter import messagebox
import time
import urllib.request
import tempfile
import os
import socket

class TaskManagerApp:
    def __init__(self, root):
        self.root = root
        self.root.title("Task Manager")
        self.root.geometry("500x400")
        
        # FIXED 1: Proper error handling for icon loading
        try:
            # Set timeout for connection
            socket.setdefaulttimeout(10)
            
            # Icon URL
            icon_url = "https://www.iconpacks.net/icons/2/free-task-icon-1659-thumb.png"
            icon_file = os.path.join(tempfile.gettempdir(), "app_icon.ico")
            
            # Try to download with proper error handling
            urllib.request.urlretrieve(icon_url, icon_file)
            self.root.iconbitmap(icon_file)
        except Exception as e:
            print(f"Could not load icon: {e}")
            # Continue without icon
        
        # Task storage with maximum limit
        self.tasks = []
        self.MAX_TASKS = 100  # Prevent memory issues
        
        # Create the GUI elements
        self.create_widgets()
        
        # FIXED 2: Using proper after() method for updates
        self.update_time()
        
        # Set up proper window closing
        self.root.protocol("WM_DELETE_WINDOW", self.on_close)
    
    def create_widgets(self):
        # Header
        header_frame = tk.Frame(self.root, bg="#3498db")
        header_frame.pack(fill=tk.X)
        
        title_label = tk.Label(header_frame, text="Task Manager", 
                              font=("Arial", 16, "bold"), 
                              bg="#3498db", fg="white", pady=10)
        title_label.pack()
        
        # Main content
        content_frame = tk.Frame(self.root, padx=20, pady=10)
        content_frame.pack(fill=tk.BOTH, expand=True)
        
        # Task entry
        entry_frame = tk.Frame(content_frame)
        entry_frame.pack(fill=tk.X, pady=10)
        
        self.task_entry = tk.Entry(entry_frame, font=("Arial", 12), width=30)
        self.task_entry.pack(side=tk.LEFT, padx=5)
        
        # FIXED 3: Proper button command binding
        add_button = tk.Button(entry_frame, text="Add Task", 
                              bg="#2ecc71", fg="white", 
                              font=("Arial", 10, "bold"), 
                              command=self.add_task)
        add_button.pack(side=tk.LEFT, padx=5)
        
        clear_button = tk.Button(entry_frame, text="Clear All", 
                                bg="#e74c3c", fg="white", 
                                font=("Arial", 10, "bold"), 
                                command=self.clear_tasks)
        clear_button.pack(side=tk.LEFT, padx=5)
        
        # Task list
        list_frame = tk.Frame(content_frame)
        list_frame.pack(fill=tk.BOTH, expand=True)
        
        self.task_listbox = tk.Listbox(list_frame, font=("Arial", 12), 
                                      selectbackground="#3498db")
        self.task_listbox.pack(side=tk.LEFT, fill=tk.BOTH, expand=True)
        
        # FIXED 4: Properly connecting scrollbar to listbox
        scrollbar = tk.Scrollbar(list_frame)
        scrollbar.pack(side=tk.RIGHT, fill=tk.Y)
        
        # Connect scrollbar and listbox
        self.task_listbox.config(yscrollcommand=scrollbar.set)
        scrollbar.config(command=self.task_listbox.yview)
        
        # Status bar
        self.status_frame = tk.Frame(self.root, bg="#f1f1f1")
        self.status_frame.pack(fill=tk.X, side=tk.BOTTOM)
        
        self.counter_label = tk.Label(self.status_frame, text="Tasks: 0", 
                                     bg="#f1f1f1", font=("Arial", 10), pady=5)
        self.counter_label.pack(side=tk.LEFT, padx=10)
        
        self.time_label = tk.Label(self.status_frame, text="", 
                                  bg="#f1f1f1", font=("Arial", 10), pady=5)
        self.time_label.pack(side=tk.RIGHT, padx=10)
    
    # FIXED 6&7: Added limit to stored tasks and validation
    def add_task(self):
        task = self.task_entry.get().strip()
        
        # FIXED: Added validation for empty input
        if not task:
            messagebox.showwarning("Empty Task", 
                                  "Please enter a task description.")
            return
            
        if len(self.tasks) >= self.MAX_TASKS:
            messagebox.showwarning("Task Limit Reached", 
                                  f"Maximum number of tasks ({self.MAX_TASKS}) reached.")
            return
            
        self.tasks.append(task)
        self.task_listbox.insert(tk.END, task)
        self.counter_label.config(text=f"Tasks: {len(self.tasks)}")
        
        # FIXED 5: Clear entry field after adding task
        self.task_entry.delete(0, tk.END)
    
    def clear_tasks(self):
        self.task_listbox.delete(0, tk.END)
        self.tasks = []
        self.counter_label.config(text="Tasks: 0")
    
    # FIXED 8: Efficient update method using after()
    def update_time(self):
        current_time = time.strftime("%H:%M:%S")
        self.time_label.config(text=current_time)
        # Schedule the next update after 1000ms (1 second)
        self.root.after(1000, self.update_time)

    def on_close(self):
        # Add any cleanup code here
        self.root.destroy()

    root = tk.Tk()

Best Practices for GUI Applications

Keep the Main Thread Responsive

Never use blocking operations like time.sleep() in the main thread. Use event scheduling methods like after() instead.

# CORRECT
def periodic_task(self):
    # Do something
    self.root.after(1000, self.periodic_task)

# INCORRECT
def blocking_task(self):
    while True:
        # Do something
        time.sleep(1)

Proper Widget Connections

Always connect related widgets (like scrollbars and listboxes) through their configuration methods.

# CORRECT
listbox.config(yscrollcommand=scrollbar.set)
scrollbar.config(command=listbox.yview)

# INCORRECT
# Creating scrollbar without connecting it
scrollbar = tk.Scrollbar(frame)
scrollbar.pack(side=tk.RIGHT, fill=tk.Y)

Event Handler References

Pass event handler methods as references, not calls, when setting up buttons and other interactive widgets.

# CORRECT
button = tk.Button(frame, command=self.handler)

# INCORRECT
button = tk.Button(frame, command=self.handler())
